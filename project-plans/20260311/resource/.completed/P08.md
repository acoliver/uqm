### 2. Fixed `res_OpenResFile` to detect directories
```rust
pub unsafe extern "C" fn res_OpenResFile(
    dir: *mut c_void,
    filename: *const c_char,
    mode: *const c_char,
) -> *mut c_void {
    if filename.is_null() || mode.is_null() {
        return ptr::null_mut();
    }

    // Check if target is a directory using uio_stat
    let mut sb: libc::stat = std::mem::zeroed();
    if uio_stat(dir, filename, &mut sb) == 0 {
        // S_ISDIR: check if st_mode has the directory bit set
        // Use libc::S_IFMT and libc::S_IFDIR macros for portability
        if (sb.st_mode & libc::S_IFMT) == libc::S_IFDIR {
            return STREAM_SENTINEL;
        }
    }

    // Not a directory or stat failed — fall through to regular open
    uio_fopen(dir, filename, mode)
}
```

Key changes:
- Calls `uio_stat` before `uio_fopen`
- Uses `libc::S_IFMT` and `libc::S_IFDIR` for portability (st_mode is u16 on macOS, u32 on Linux)
- Returns `STREAM_SENTINEL` (!0) for directories
- Falls through to normal open for non-directories or stat failures (graceful degradation)

### 3. Updated `LoadResourceFromPath` to guard against sentinel and zero-length
```rust
pub unsafe extern "C" fn LoadResourceFromPath(
    pathname: *const c_char,
    load_fn: Option<ResourceLoadFileFun>,
) -> *mut c_void {
    if pathname.is_null() {
        return ptr::null_mut();
    }

    let load_fn = match load_fn {
        Some(f) => f,
        None => return ptr::null_mut(),
    };

    let mode = b"rb\0".as_ptr() as *const c_char;
    let fp = res_OpenResFile(contentDir, pathname, mode);
    
    // Reject null, sentinel (directory), and check for zero-length files
    if fp.is_null() || fp == STREAM_SENTINEL {
        return ptr::null_mut();
    }

    // Get file length
    let length = LengthResFile(fp);
    if length == 0 {
        log::warn!("LoadResourceFromPath: zero-length file, closing and returning null");
        res_CloseResFile(fp);
        return ptr::null_mut();
    }

    // Set global file name for C callers
    _cur_resfile_name = pathname;

    let result = load_fn(fp, length as u32);

    // Clear global file name
    _cur_resfile_name = ptr::null();

    res_CloseResFile(fp);
    result
}
```

Key changes:
- Now calls `res_OpenResFile` instead of `uio_fopen` directly (gets directory sentinel check)
- Explicitly rejects `STREAM_SENTINEL` (directories)
- Uses `LengthResFile` instead of manual seeking
- Checks for zero-length files and closes handle before returning null (prevents handle leaks)
- Now uses `res_CloseResFile` for consistency
